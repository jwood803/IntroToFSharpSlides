My first look at F#


let rec cast<'a> (myList: obj list) =
	match myList with
	| head::tail ->
		match head with
		| :? 'a as a -> a::(cast tail)
		| _ -> cast tail
	| [] -> []

let getHomePhone phoneNumber =
	match phoneNumber.Home with
	| Some number -> printfn "Number is %s" number
	| None -> printfn "No number provided"
